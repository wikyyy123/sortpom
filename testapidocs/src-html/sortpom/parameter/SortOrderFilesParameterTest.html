001package sortpom.parameter;
002
003import org.junit.Rule;
004import org.junit.Test;
005import org.junit.rules.ExpectedException;
006import sortpom.util.SortPomImplUtil;
007
008import static org.hamcrest.Matchers.endsWith;
009
010public class SortOrderFilesParameterTest {
011
012    @Rule
013    public ExpectedException thrown = ExpectedException.none();
014
015    @Test
016    public final void incorrectCustomSortOrderShouldThrowException() throws Exception {
017        thrown.expect(RuntimeException.class);
018        thrown.expectMessage(endsWith("VERYdifferentOrder.xml in classpath"));
019        SortPomImplUtil.create()
020                .defaultOrderFileName("difforder/VERYdifferentOrder.xml")
021                .testFiles("/full_unsorted_input.xml", "/sortOrderFiles/sorted_differentOrder.xml");
022    }
023
024    @Test
025    public final void incorrectPredefinedSortOrderShouldThrowException() throws Exception {
026        thrown.expect(IllegalArgumentException.class);
027        thrown.expectMessage("Cannot find abbie_normal_brain.xml among the predefined plugin resources");
028        SortPomImplUtil.create()
029                .predefinedSortOrder("abbie_normal_brain")
030                .lineSeparator("\n")
031                .testFiles("/full_unsorted_input.xml",
032                        "/sortOrderFiles/sorted_default_0_4_0.xml");
033    }
034
035    @Test
036    public final void incorrectCustomSortOrderShouldThrowException2() throws Exception {
037        thrown.expect(RuntimeException.class);
038        thrown.expectMessage(endsWith("VERYdifferentOrder.xml in classpath"));
039        SortPomImplUtil.create()
040                .defaultOrderFileName("difforder/VERYdifferentOrder.xml")
041                .testVerifyXmlIsOrdered("/sortOrderFiles/sorted_differentOrder.xml");
042    }
043
044    @Test
045    public final void incorrectPredefinedSortOrderShouldThrowException2() throws Exception {
046        thrown.expect(IllegalArgumentException.class);
047        thrown.expectMessage("Cannot find abbie_normal_brain.xml among the predefined plugin resources");
048        SortPomImplUtil.create()
049                .predefinedSortOrder("abbie_normal_brain")
050                .lineSeparator("\n")
051                .testVerifyXmlIsOrdered("/sortOrderFiles/sorted_default_0_4_0.xml");
052    }
053
054}
